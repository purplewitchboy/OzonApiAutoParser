function loadStockFBO() {
  Logger.log('=== loadStockFBO start ===');

  const settings = getSettings();
  const ss = SpreadsheetApp.getActiveSpreadsheet();

  const sheetName = 'Склад FBO';
  let sheet = ss.getSheetByName(sheetName);
  if (!sheet) {
    sheet = ss.insertSheet(sheetName);
  }

  sheet.clear();

  sheet.getRange(1, 1, 1, 10).setValues([[
    'Кабинет',
    'Дата',
    'SKU',
    'Артикул поставщика',
    'Количество товара, доступное к продаже на Ozon (остаток)',
    'Количество товара, указанное в подтверждённых будущих поставках',
    'Количество товара, зарезервированное для покупки, возврата и перевозки между складами',
    'Название склада, где находится',
    'На сколько дней хватит остатка товара с учётом среднесуточных продаж по кластеру',
    'Кластер'
  ]]);

  // ---------- STOCK ON WAREHOUSES ----------
  const url = 'https://api-seller.ozon.ru/v2/analytics/stock_on_warehouses';

  const payload = {
    filter: {
      since: toIso(settings.dateFrom),
      to: toIso(settings.dateTo)
    },
    warehouse_type: 'fbo',
    limit: 1000,
    offset: 0
  };

  const options = {
    method: 'post',
    contentType: 'application/json',
    headers: {
      'Client-Id': String(settings.sellerClientId),
      'Api-Key': settings.sellerApiKey
    },
    payload: JSON.stringify(payload),
    muteHttpExceptions: true
  };

  const response = UrlFetchApp.fetch(url, options);
  const code = response.getResponseCode();
  const text = response.getContentText();

  Logger.log('Stock FBO HTTP CODE: ' + code);

  if (code !== 200) {
    Logger.log(text);
    throw new Error('Stock FBO error ' + code + ': ' + text);
  }

  const json = JSON.parse(text);
  const rowsData = json.result?.rows || [];

  // ---------- ANALYTICS STOCKS ----------
  const skuList = [...new Set(
    rowsData.map(r => String(r.sku)).filter(Boolean)
  )];

  const daysMap = {};     // sku|warehouse -> idc_cluster
  const clusterMap = {};  // sku|warehouse -> cluster_name

  if (skuList.length) {
    const stocksUrl = 'https://api-seller.ozon.ru/v1/analytics/stocks';

    const stocksPayload = {
      skus: skuList
    };

    const stocksResp = UrlFetchApp.fetch(stocksUrl, {
      method: 'post',
      contentType: 'application/json',
      headers: {
        'Client-Id': String(settings.sellerClientId),
        'Api-Key': settings.sellerApiKey
      },
      payload: JSON.stringify(stocksPayload),
      muteHttpExceptions: true
    });

    if (stocksResp.getResponseCode() === 200) {
      const stocksJson = JSON.parse(stocksResp.getContentText());

      (stocksJson.items || []).forEach(it => {
        const wh = normalizeWarehouseName(it.warehouse_name);
        const key = `${it.sku}|${wh}`;

        daysMap[key] = it.idc_cluster ?? '';
        clusterMap[key] = it.cluster_name ?? '';
      });
    } else {
      Logger.log('Analytics stocks error: ' + stocksResp.getContentText());
    }
  }
  // ---------- END ANALYTICS STOCKS ----------

  // ---------- OUTPUT ----------
  const rows = [];

  rowsData.forEach(row => {
    const wh = normalizeWarehouseName(row.warehouse_name);
    const key = `${row.sku}|${wh}`;

    let daysOnHand = daysMap[key] ?? '';
    let clusterName = clusterMap[key] ?? '';

    // fallback: если не нашли по складу — ищем просто по SKU
    if (daysOnHand === '' && clusterName === '') {
      const fallbackKey = Object.keys(daysMap).find(k => k.startsWith(row.sku + '|'));
      if (fallbackKey) {
        daysOnHand = daysMap[fallbackKey] ?? '';
        clusterName = clusterMap[fallbackKey] ?? '';
      }
    }

    rows.push([
      settings.cabinetName,
      Utilities.formatDate(
        new Date(settings.dateTo),
        Session.getScriptTimeZone(),
        'yyyy-MM-dd'
      ),
      row.sku || '',
      row.item_code || '',
      row.free_to_sell_amount,
      row.promised_amount,
      row.reserved_amount,
      row.warehouse_name || '',
      daysOnHand,
      clusterName
    ]);
  });

  if (rows.length) {
    sheet.getRange(2, 1, rows.length, rows[0].length).setValues(rows);
  }

  Logger.log(`=== loadStockFBO done. Rows: ${rows.length} ===`);
}


// ---------------- NORMALIZE WAREHOUSE NAME ----------------
function normalizeWarehouseName(name) {
  if (!name) return '';

  let n = String(name).trim().toUpperCase();

  // базовая очистка
  n = n.replace(/_/g, ' ');
  n = n.replace(/\s+/g, ' ');
  n = n.replace(/Ё/g, 'Е');

  // алиасы проблемных складов
  const aliases = {
    'НОВОРОССИЙСК МРФЦ': 'НОВОРОССИЙСК',
    'НОВОРОССИЙСК МРФЦ СКЛАД': 'НОВОРОССИЙСК'
  };

  return aliases[n] || n;
}
