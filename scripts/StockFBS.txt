function loadStockFBS() {
  Logger.log('=== loadStockFBS start ===');

  const settings = getSettings();
  const ss = SpreadsheetApp.getActiveSpreadsheet();

  const sheetName = 'Склад FBS';
  let sheet = ss.getSheetByName(sheetName);
  if (!sheet) {
    sheet = ss.insertSheet(sheetName);
  }

  sheet.clear();

  sheet.getRange(1, 1, 1, 6).setValues([[
    'Кабинет',
    'Дата',
    'Количество',
    'Зарезервировано',
    'sku',
    'Название склада'
  ]]);

  const url = 'https://api-seller.ozon.ru/v2/analytics/stock_on_warehouses';

  const payload = {
    filter: {
      since: toIso(settings.dateFrom),
      to: toIso(settings.dateTo)
    },
    warehouse_type: 'fbs',
    limit: 1000,
    offset: 0
  };

  const options = {
    method: 'post',
    contentType: 'application/json',
    headers: {
      'Client-Id': String(settings.sellerClientId),
      'Api-Key': settings.sellerApiKey
    },
    payload: JSON.stringify(payload),
    muteHttpExceptions: true
  };

  const response = UrlFetchApp.fetch(url, options);
  const code = response.getResponseCode();
  const text = response.getContentText();

  Logger.log('Stock FBS HTTP CODE: ' + code);

  if (code !== 200) {
    Logger.log(text);
    throw new Error('Stock FBS error ' + code + ': ' + text);
  }

  const json = JSON.parse(text);
  const rows = [];

  (json.result?.rows || []).forEach(row => {
    rows.push([
      settings.cabinetName,
      Utilities.formatDate(
        new Date(settings.dateTo),
        Session.getScriptTimeZone(),
        'yyyy-MM-dd'
      ),
      row.total || 0,        // Количество
      row.reserved || 0,     // Зарезервировано
      row.sku || '',         // sku
      row.warehouse_name || ''
    ]);
  });

  if (rows.length) {
    sheet.getRange(2, 1, rows.length, rows[0].length).setValues(rows);
  }

  Logger.log(`=== loadStockFBS done. Rows: ${rows.length} ===`);
}
