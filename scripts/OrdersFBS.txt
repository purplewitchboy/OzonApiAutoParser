function loadOrdersFBS() {
  Logger.log('=== loadOrdersFBS start ===');

  const settings = getSettings();
  const ss = SpreadsheetApp.getActiveSpreadsheet();

  const sheetName = 'Заказы FBS';
  let sheet = ss.getSheetByName(sheetName);
  if (!sheet) {
    sheet = ss.insertSheet(sheetName);
  }

  sheet.clear();

  sheet.getRange(1, 1, 1, 14).setValues([[
    'Кабинет',
    'Номер заказа',
    'Номер отправления',
    'Принят в обработку',
    'Статус',
    'Сумма отправления',
    'Код валюты отправления',
    'OZON id',
    'Артикул',
    'Итоговая стоимость товара',
    'Код валюты товара',
    'Стоимость товара для покупателя',
    'Код валюты покупателя',
    'Количество'
  ]]);

  const url = 'https://api-seller.ozon.ru/v3/posting/fbs/list';

  const payload = {
    filter: {
      since: toIso(settings.dateFrom),
      to: toIso(settings.dateTo)
    },
    limit: 1000,
    offset: 0
  };

  const options = {
    method: 'post',
    contentType: 'application/json',
    headers: {
      'Client-Id': String(settings.sellerClientId),
      'Api-Key': settings.sellerApiKey
    },
    payload: JSON.stringify(payload),
    muteHttpExceptions: true
  };

  const response = UrlFetchApp.fetch(url, options);
  const code = response.getResponseCode();
  const text = response.getContentText();

  Logger.log('Orders FBS HTTP CODE: ' + code);

  if (code !== 200) {
    Logger.log(text);
    throw new Error('Orders FBS error ' + code + ': ' + text);
  }

  const json = JSON.parse(text);
  Logger.log('FBS postings count: ' + (json.result?.postings?.length || 0));
  const rows = [];

  (json.result?.postings || []).forEach(posting => {
    (posting.products || []).forEach(product => {
      const price = Number(product.price || 0);
      const quantity = Number(product.quantity || 0);
      const currency = product.currency_code || '';

      rows.push([
        settings.cabinetName,                         // Кабинет
        posting.order_number,                         // Номер заказа
        posting.posting_number,                       // Номер отправления
        posting.in_process_at,                        // Принят в обработку
        posting.status,                               // Статус
        posting.financial_data?.posting_services
          ?.marketplace_service_item_fulfillment || '', // Сумма отправления
        posting.financial_data?.posting_services
          ?.currency || '',                           // Код валюты отправления
        product.ozon_id,                              // OZON id
        product.offer_id,                             // Артикул
        price * quantity || '',                       // Итоговая стоимость товара
        currency,                                     // Код валюты товара
        price || '',                                  // Стоимость товара для покупателя
        currency,                                     // Код валюты покупателя
        quantity                                      // Количество
      ]);
    });
  });

  if (rows.length) {
    sheet.getRange(2, 1, rows.length, rows[0].length).setValues(rows);
  }

  Logger.log(`=== loadOrdersFBS done. Rows: ${rows.length} ===`);
}
